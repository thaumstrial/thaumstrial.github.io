$$
\cos(\alpha + \beta) = \frac{OE}{OC}
$$
$$
OE = OD - ED
$$
$$
ED = HB
$$
$$
\begin{align*}
\cos(\alpha) &= \frac{OD}{OB} \\
OB\cos(\alpha) &= OD
\end{align*}
$$
$$
\begin{align*}
\cos(\beta) &= \frac{OB}{OC} \\
OB\ &= OC\cos(\beta)
\end{align*}
$$
$$
\begin{align*}
OC\cos(\beta)\cos(\alpha) &= OD
\end{align*}
$$
$$
\begin{align*}
\cos(\alpha) &= \frac{HB}{CB} \\
CB\cos(\alpha) &= ED
\end{align*}
$$
$$
\begin{align*}
\sin(\beta) &= \frac{CB}{OC} \\
CB &= OC\sin(\beta)
\end{align*}
$$
$$
OC\sin(\beta)\cos(\alpha) = ED
$$
$$
\begin{align*}
\cos(\alpha + \beta) &= \frac{OC\cos(\beta)\cos(\alpha) + OC\sin(\beta)\cos(\alpha)}{OC} \\
&= \cos(\alpha)\cos(\beta) + \sin(\alpha)\cos(\beta) \\
\end{align*}
$$
